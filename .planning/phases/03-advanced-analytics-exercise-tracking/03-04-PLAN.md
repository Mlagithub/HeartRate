```html
<script lang="ts">
  import { onMount } from 'svelte';
  import { invoke } from '@tauri-apps/api/core';
  import { format } from 'date-fns';

  // Types matching Rust backend
  interface ExerciseStats { ... }
  interface HeartRateZones { ... }
  interface ExerciseTypeStats { ... }

  let stats: ExerciseStats | null = null;
  let typeStats: ExerciseTypeStats[] = [];
  let isLoading = true;
  let error: string | null = null;

  onMount(async () => {
    await loadExerciseStats();
  });

  async function loadExerciseStats() {
    isLoading = true;
    error = null;
    try {
      stats = await invoke<ExerciseStats>('get_exercise_statistics');
      typeStats = await invoke<ExerciseTypeStats[]>('get_exercise_type_statistics');
    } catch (e) {
      error = String(e);
    }
    isLoading = false;
  }
</script>

<div class="exercise-tab">

</div>
```

```typescript
interface HRVResult {
  hrv_value: number;
  is_estimated: boolean;
  confidence: string;
  data_points: number;
  period_start: number;
  period_end: number;
}

let hrvData: HRVResult | null = null;
let hrvLoading = false;
let hrvError: string | null = null;

async function loadHRV() {
  hrvLoading = true;
  hrvError = null;
  try {
    hrvData = await invoke<HRVResult | null>('get_hrv_estimate', {
      startTime: null,
      endTime: null
    });
  } catch (e) {
    hrvError = String(e);
  }
  hrvLoading = false;
}
```
